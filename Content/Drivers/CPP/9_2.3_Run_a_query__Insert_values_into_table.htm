Run a query - Insert values into table

Example
string  statement  =  "insert into table_name(int_column) values (5), (6), (7), (8)";  
sqc.new_query(statement);  
sqc.execute_query();  
sqc.finish_query();  

OR  

run_direct_query(&sqc,  "insert into table_name(int_column)  values  (5),  (6),  (7),  (8)");
